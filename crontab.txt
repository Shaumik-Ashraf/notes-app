# Notes App — host crontab
#
# Install with:
#   crontab /home/ubuntu/notes-app/crontab.txt
#
# Verify with:
#   crontab -l
#
# Rake tasks are run by starting a one-off container from the already-built
# app image. The container shares the same .env file and notes_storage volume
# as the running app service, then exits immediately after the task completes.
#
# Flags:
#   --rm          remove the container after it exits
#   --no-deps     do not start the db service (already running)
#   --entrypoint=""  bypass bin/docker-entrypoint so db:prepare is not re-run

# Delete notes older than PURGE_NOTES_AFTER_DAYS days (default: 30) — daily at 02:00
0 2 * * * cd /home/ubuntu && docker run --rm --no-deps --entrypoint="" ghcr.io/shaumik-ashraf/notes-app ./bin/rails notes:purge
